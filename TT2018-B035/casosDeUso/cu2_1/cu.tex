	\begin{UseCase}{LR-CU2.1}{Visualizar Información de Zona Turística}{
		
		Permite realizar la consulta de la información de alguna zona turística que sea del interés del \cdtRef{actor:turista}{Turista}. La zona turística puede ser la zona actual o alguna otra zona turística dentro del sistema.\\
		
	}
		\UCitem{Versión}{\color{Gray}1.0}
		\UCitem{Actor}{\cdtRef{actor:turista}{Turista}}
		\UCitem{Propósito}{Visualizar el listado de las zonas turísticas.}
		\UCitem{Entradas}{Zona turística}
		\UCitem{Origen}{\ioSistema}
		\UCitem{Salidas}{Información sobre la zona turística}
		\UCitem{Precondiciones}{Haber seleccionado una zona turística.}
		\UCitem{Postcondiciones}{Podrá visualizar la información de la zona turística.}
		\UCitem{Reglas de negocio}{Ninguna}
		\UCitem{Errores}{Ninguno}
		\UCitem{Tipo}{Extiende de \cdtIdRef{LR-CU2}{Obtener Zonas Turísticas}}
	\end{UseCase}
%--------------------------------------
	\begin{UCtrayectoria} 
		\UCpaso [\UCactor] Solicita visualizar la información seleccionando una zona turística en la pantalla \IUref{LR-IU2}{Zonas turísticas}.
		
		\UCpaso Obtiene el nombre y la descripción de la zona turística seleccionada.
		
		\UCpaso Muestra la pantalla \IUref{LR-IU2.1a}{Zona Turística}. \refTray{A}
		
		\UCpaso [\UCactor] Solicita ver el mapa de la zona turística tocando el botón \IUbutton{Ver mapa}.
		
		\UCpaso Obtiene las coordenadas de la zona turística.
		
		\UCpaso Obtiene las coordenadas de los servicios turísticos que pertenecen a la zona turística seleccionada.
		
		\UCpaso \label{LR-CU2.1:Pantalla2} Muestra la pantalla \IUref{LR-IU2.1b}{Mapa de Zona Turística}.
		
	\end{UCtrayectoria}	

	\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{Cuando el actor requiere ver los serivicios turísticos de la zona.}
		
		\UCpaso [\UCactor] Solicita ver los servicios turísticos que pertenecen a la zona turística seleccionada.
		
		\UCpaso \label{LR-CU2.1:Pantalla1} Continua en el caso de uso \cdtIdRef{SE-CU2.1.2}{Consultar Información de Servicios Turísticos}
		
	\end{UCtrayectoriaA}
	
	\subsection{Puntos de extensión}
	\UCExtenssionPoint{El turista requiere visualizar la información de los servicios turísticos.}{En el paso \ref{LR-CU2.1:Pantalla1} de la trayectoria alterna \textbf{A}.}{\cdtIdRef{LR-CU2.1.2}{Consultar información de servicios turísticos}.}
	
	\UCExtenssionPoint{El turista requiere generar una ruta turística a partir de su ubicación actual.}{En el paso \ref{LR-CU2.1:Pantalla2} de la trayectoria principal.}{\cdtIdRef{LR-CU2.1.1}{Generar ruta turística}.}
