\begin{BusinessRule}{BR-001}{Inicio de sesión con Apple ID}
	{Restricción de operación}
	{Obligatorio}
	\BRitem{Descripción}{Para poder crear una cuenta dentro de la aplicación móvil deberá contarse con un Apple ID, el mismo que será utilizado para descargar la aplicación de la tienda AppStore de iPhone.}
	
\end{BusinessRule}

\begin{BusinessRule}{BR-002}{Longitud válida para nombre de usuario}
	{Restricción de operación}
	{Obligatorio}
	\BRitem{Descripción}{El nombre de usuario debe tener al entre 3 y 16 caracteres, se puede introducir letras números y guiones. No puede tener otros símbolos.

	}
	
\end{BusinessRule}
