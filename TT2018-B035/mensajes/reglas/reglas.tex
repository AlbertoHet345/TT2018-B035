\begin{BusinessRule}{BR-001}{Longitud válida para contraseñas}
	{Restricción de operación}
	{Obligatorio}
	\BRitem{Descripción}{La contraseña debe tener al entre 8 y 16 caracteres, al menos un dígito, al menos una minúscula y al menos una mayúscula. No puede tener otros símbolos.
	Expresión Regular:  }
	
\end{BusinessRule}

\begin{BusinessRule}{BR-002}{Longitud válida para nombre de usuario}
	{Restricción de operación}
	{Obligatorio}
	\BRitem{Descripción}{El nombre de usuario debe tener al entre 3 y 16 caracteres, se puede introducir letras números y guiones. No puede tener otros símbolos.
		Expresión Regular: 
	}
	
\end{BusinessRule}
